//sql for liteCMS

CREATE TABLE lite_post (
    id INTEGER PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(255) NOT NULL,
    body TEXT(30000) NOT NULL,
    author VARCHAR(255) NOT NULL,
    likes INTEGER DEFAULT(0) NOT NULL,
    status INTEGER DEFAULT(0) NOT NULL,
)

CREATE TABLE lite_comment(
    id INTEGER NOT NULL AUTO_INCREMENT PRIMARY KEY,
    post INTEGER NOT NULL,
    commenter VARCHAR(255) NOT NULL
)

CREATE TABLE lite_tags(
    id INTEGER NOT NULL PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(255) NOT NULL,
    color_code VARCHAR(9) NOT NULL
)

ALTER TABLE `lite_comment` ADD FOREIGN KEY (`post`) REFERENCES `lite_post`(`id`) ON DELETE CASCADE ON UPDATE CASCADE;
